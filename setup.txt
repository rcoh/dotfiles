#!/bin/bash
sudo apt-get install xmonad trayer suckless-tools cabal-install xmobar
# For volume control
cabal update
cabal install xmonad-extras 

# For SSH keyring:
echo 'export SSH_AUTH_SOCK="$GNOME_KEYRING_CONTROL/ssh"' >> ~/.profile

# Modify /usr/share/xsessions/xmonad.desktop to run /home/russell/.xmonadrc

# Link spotify.sh in usr/bin as spotifyctl
# Link spotify.py in /usr/bin/spotifyinfo
# Link notify.py in /usr/bin/notify
HERE=`pwd`
function linkhome() {
	ln -sf $HERE/$1 ~/$1
}

linkhome .xmonad
linkhome .xmonadrc
linkhome .xmobarrc
